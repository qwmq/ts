const NOW = new Date();

function daysDiff(dateStr){
  if(!dateStr) return 9999;
  return (NOW - new Date(dateStr)) / (1000*60*60*24);
}

function getPrice(it){
  if(it.price) return it.price;
  const kinds = new Set(it.id.replace(/\D/g,'')).size;
  if(kinds<=3) return 2000;
  if(kinds===4) return 1800;
  return 1500;
}

function isArchived(it){
  return it.status==='sold' && it.soldDate && daysDiff(it.soldDate)>=5;
}

function isNew(it){
  return it.addedDate && daysDiff(it.addedDate)<=3;
}

function displayId(id){ return id.length===6 ? '@'+id : id; }
function twitterUrl(id){ return 'https://twitter.com/'+id; }
function esc(s){ return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;'); }

let sortMode='default', searchQ='', filterMode='all', priceFilter='all';

function render(){
  let list = items
    .filter(it=>!isArchived(it))
    .map((it,i)=>({...it, _i:i, price:getPrice(it), isNew:isNew(it)}));

  if(searchQ){
    const q=searchQ.toLowerCase();
    list=list.filter(it=>it.id.toLowerCase().includes(q));
  }
  if(filterMode==='available') list=list.filter(it=>it.status==='available');
  if(filterMode==='sold')      list=list.filter(it=>it.status==='sold');

  if(priceFilter==='2500') list=list.filter(it=>it.price>=2500);
  else if(priceFilter!=='all') list=list.filter(it=>it.price===parseInt(priceFilter));

  // available を前、sold を後ろに
  list.sort((a,b)=>{
    if(a.status===b.status) return 0;
    return a.status==='available' ? -1 : 1;
  });

  // available内でソート
  if(sortMode==='default'){
    // available同士はランダム、sold同士はランダム
    const avail=list.filter(x=>x.status==='available').sort(()=>Math.random()-.5);
    const sold =list.filter(x=>x.status==='sold').sort(()=>Math.random()-.5);
    list=[...avail,...sold];
  } else if(sortMode==='id'){
    list.sort((a,b)=>{
      if(a.status!==b.status) return a.status==='available'?-1:1;
      return a.id.localeCompare(b.id);
    });
  } else if(sortMode==='asc'){
    list.sort((a,b)=>{
      if(a.status!==b.status) return a.status==='available'?-1:1;
      return a.price-b.price;
    });
  } else if(sortMode==='desc'){
    list.sort((a,b)=>{
      if(a.status!==b.status) return a.status==='available'?-1:1;
      return b.price-a.price;
    });
  }

  const grid=document.getElementById('grid');
  grid.innerHTML='';
  const empty=document.getElementById('empty');

  if(list.length===0){empty.style.display='block';}
  else{
    empty.style.display='none';
    list.forEach((it,delay)=>{
      const avail=it.status==='available';
      const card=document.createElement('div');
      card.className='card '+(avail?'available':'is-sold');
      card.style.animationDelay=(delay*0.012)+'s';
      card.innerHTML=`
        <div class="card-top">
          <div class="card-id">${esc(displayId(it.id))}</div>
          <div style="display:flex;flex-direction:column;gap:3px;align-items:flex-end">
            <span class="badge ${avail?'badge-available':'badge-sold'}">${avail?'✓ available':'✕ sold'}</span>
            ${it.isNew && avail ? '<span class="badge badge-new">NEW</span>' : ''}
          </div>
        </div>
        <div class="card-price">¥${it.price.toLocaleString()}<span>JPY</span></div>
        <div class="card-btns" style="flex-direction:column;">
          <button class="btn-buy" onclick="openBuyModal('${esc(displayId(it.id))}')">購入する</button>
          <a class="btn-twitter" href="${twitterUrl(it.id)}" target="_blank" rel="noopener">Twitterで見る</a>
        </div>`;
      grid.appendChild(card);
    });
  }

  const visible=items.filter(it=>!isArchived(it));
  document.getElementById('statTotal').textContent=visible.length;
  document.getElementById('statAvail').textContent=visible.filter(i=>i.status==='available').length;
  document.getElementById('statSold').textContent=visible.filter(i=>i.status==='sold').length;
}
